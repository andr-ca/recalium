```python
"""Conflict and duplicate detection for derived memory facts.

CANM-06: Flags contradictory or duplicate facts across sources by comparing
embedding vectors using pgvector cosine distance.

SECURITY: All queries filter source_status = 'active' — suppressed data is excluded.
"""
from __future__ import annotations

import logging
import uuid

from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession

from app.domain.derived_memory.models import ConflictGroup

logger = logging.getLogger(__name__)

# Cosine distance threshold for duplicate detection.
# pgvector <=> returns cosine distance: 0.0 = identical, 2.0 = opposite.
# 0.15 ≈ cosine similarity 0.85 — highly similar content.
# Tune empirically if false positives occur.
DUPLICATE_DISTANCE_THRESHOLD = 0.15

# Maximum duplicate candidates to examine per new embedding.
_MAX_CANDIDATES = 10


async def find_duplicate_candidates(
    session: AsyncSession,
    embedding: list[float],
    exclude_id: uuid.UUID,
) -> list[uuid.UUID]:
    """Find embeddings that are near-duplicates of the given vector.

    Uses pgvector <=> (cosine distance) with HNSW index on embeddings table.
    Filters source_status = 'active' — suppressed embeddings are excluded (cascade contract).
    Excludes exclude_id (the embedding just written — avoid self-match).

    Returns list of up to _MAX_CANDIDATES matching embedding IDs.
    """
    result = await session.execute(
        text("""
            SELECT id
            FROM embeddings
            WHERE id != :exclude_id
              AND source_status = 'active'
              AND embedding <=> :vec < :threshold
            ORDER BY embedding <=> :vec
            LIMIT :max_candidates
        """),
        {
            "vec": str(embedding),
            "exclude_id": str(exclude_id),
            "threshold": DUPLICATE_DISTANCE_THRESHOLD,
            "max_candidates": _MAX_CANDIDATES,
        },
    )
    rows = result.fetchall()
    logger.debug(
        "Conflict scan: found %d candidates near embedding %s",
        len(rows), exclude_id,
    )
    return [uuid.UUID(str(row.id)) for row in rows]


async def create_conflict_group(
    session: AsyncSession,
    group_type: str = "duplicate",
) -> ConflictGroup:
    """Create a new conflict group record.

    CANM-06: group_type values: "duplicate" | "contradiction" | "overlap"
    For v1 conflict detection, only "duplicate" is used.

    The caller links facts to this group by setting Fact.conflict_group_id.
    """
    group = ConflictGroup(group_type=group_type)
    session.add(group)
    await session.commit()
    await session.refresh(group)
    logger.info("Created conflict group id=%s type=%s", group.id, group_type)
    return group
```
