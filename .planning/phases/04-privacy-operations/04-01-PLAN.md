<objective>
This plan implements the deletion cascade service (PRIV-01, PRIV-02) and the authentication middleware for non-localhost exposure (PRIV-06). The cascade service marks a raw archive item as deleted and synchronously propagates `source_status='source_removed'` to all derived tables in a single transaction. Canonical memory from the deleted source is NOT deleted — it is marked `source_status='source_removed'` + `status='required_review'` so users can review it. An `archive_delete` audit event is written. The auth middleware ensures that when the server is exposed beyond localhost, all API and MCP endpoints require a bearer token configured in `.env`.
</objective>

  ```python
  from starlette.middleware.base import BaseHTTPMiddleware
  from starlette.requests import Request as StarletteRequest
  from starlette.responses import JSONResponse

  # Paths that are always exempt from auth (health check + OpenAPI docs + static files)
  _AUTH_EXEMPT_PREFIXES = ("/health", "/api/docs", "/api/redoc", "/openapi.json")
  _AUTH_REQUIRED_PREFIXES = ("/api/", "/mcp/")


  class AuthMiddleware(BaseHTTPMiddleware):
      """Bearer token auth for non-localhost deployments (PRIV-06).

      Only active when app_bind_host != '127.0.0.1'.
      Exempt: /health, /api/docs, /api/redoc, /openapi.json, and all static files.
      Required: all /api/* and /mcp/* endpoints.
      """

      async def dispatch(self, request: StarletteRequest, call_next):
          settings = get_settings()
          if not settings.requires_auth:
              return await call_next(request)

          path = request.url.path

          # Check if this path requires auth
          requires_auth = any(path.startswith(p) for p in _AUTH_REQUIRED_PREFIXES)
          is_exempt = any(path.startswith(p) for p in _AUTH_EXEMPT_PREFIXES)

          if requires_auth and not is_exempt:
              auth_header = request.headers.get("Authorization", "")
              expected = f"Bearer {settings.app_auth_bearer}"
              if not auth_header or auth_header != expected:
                  return JSONResponse(
                      {"detail": "Authentication required"},
                      status_code=401,
                  )

          return await call_next(request)
  ```

  ```python
  """Archive domain service — deletion cascade.

  PRIV-01: Deletion cascade immediately suppresses all derived data.
  PRIV-02: Canonical memory is marked source_removed + required_review, NOT deleted.

  CASCADE ORDER (all in one transaction):
    1. Set raw_archive.deleted_at = now()
    2. UPDATE summaries SET source_status='source_removed'
    3. UPDATE facts SET source_status='source_removed'
    4. UPDATE embeddings SET source_status='source_removed'
    5. UPDATE fts_entries SET source_status='source_removed'
    6. UPDATE conflict_groups SET source_status='source_removed' (via facts FK)
    7. UPDATE review_queue_items SET source_status='source_removed' (via conflict_groups)
    8. UPDATE canonical_memory SET source_status='source_removed', status='required_review'
    9. Write audit event archive_delete
  """
  from __future__ import annotations

  import logging
  import uuid
  from datetime import datetime, timezone

  from sqlalchemy import select, update, text
  from sqlalchemy.ext.asyncio import AsyncSession

  from app.domain.archive.models import RawArchiveItem
  from app.domain.audit.models import AuditEvent

  logger = logging.getLogger(__name__)


  class ArchiveItemNotFoundError(Exception):
      """Raised when the archive item does not exist or is already deleted."""


  async def cascade_delete_archive_item(
      session: AsyncSession,
      archive_id: uuid.UUID,
      actor: str = "user_ui",
  ) -> None:
      """Soft-delete a raw archive item and cascade source_removed to all derived data.

      Atomic: all updates committed together or none committed.
      Idempotent: calling twice on an already-deleted item raises ArchiveItemNotFoundError.

      PRIV-01: Derived data (summaries, facts, embeddings, fts_entries) marked source_removed.
      PRIV-02: Canonical memory marked source_removed + required_review (NOT deleted).
      """
      # 1. Fetch and validate the archive item
      result = await session.execute(
          select(RawArchiveItem).where(
              RawArchiveItem.id == archive_id,
              RawArchiveItem.deleted_at.is_(None),
          )
      )
      item = result.scalar_one_or_none()
      if item is None:
          raise ArchiveItemNotFoundError(
              f"Archive item {archive_id} not found or already deleted"
          )

      now = datetime.now(timezone.utc)
      archive_id_str = str(archive_id)

      # 2. Mark raw archive as deleted
      item.deleted_at = now

      # 3-5. Mark summaries, facts, embeddings, fts_entries as source_removed
      for table in ("summaries", "facts", "embeddings", "fts_entries"):
          await session.execute(
              text(
                  f"UPDATE {table} SET source_status = 'source_removed' "
                  "WHERE raw_archive_id = :archive_id AND source_status = 'active'"
              ),
              {"archive_id": archive_id_str},
          )

      # 6. Mark conflict_groups associated with this archive's facts as source_removed
      await session.execute(
          text(
              "UPDATE conflict_groups SET source_status = 'source_removed' "
              "WHERE id IN ("
              "  SELECT DISTINCT conflict_group_id FROM facts "
              "  WHERE raw_archive_id = :archive_id AND conflict_group_id IS NOT NULL"
              ") AND source_status = 'active'"
          ),
          {"archive_id": archive_id_str},
      )

      # 7. Mark review_queue_items linked to those conflict_groups as source_removed
      await session.execute(
          text(
              "UPDATE review_queue_items SET source_status = 'source_removed' "
              "WHERE conflict_group_id IN ("
              "  SELECT id FROM conflict_groups WHERE source_status = 'source_removed'"
              ") AND source_status = 'active'"
          ),
      )

      # 8. Mark canonical_memory as source_removed + required_review (PRIV-02: DO NOT DELETE)
      await session.execute(
          text(
              "UPDATE canonical_memory "
              "SET source_status = 'source_removed', status = 'required_review' "
              "WHERE raw_archive_id = :archive_id AND source_status = 'active'"
          ),
          {"archive_id": archive_id_str},
      )

      # 9. Write audit event
      audit_event = AuditEvent(
          event_type="archive_delete",
          raw_archive_id=archive_id,
          actor=actor,
          operation_metadata={
              "archive_id": archive_id_str,
              "deleted_at": now.isoformat(),
          },
      )
      session.add(audit_event)

      # Commit all changes atomically
      await session.commit()

      # 10. Invalidate retrieval cache (best-effort — don't fail deletion if cache fails)
      try:
          from app.domain.retrieval.service import invalidate_cache  # noqa: PLC0415
          invalidate_cache()
      except Exception as e:
          logger.warning("Could not invalidate retrieval cache after deletion: %s", e)

      logger.info(
          "Cascade delete complete for archive_id=%s actor=%s",
          archive_id_str, actor,
      )
  ```
