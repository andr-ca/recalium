"""Phase 4 integration tests: Privacy, Backup/Restore, Onboarding, Telemetry.

Covers requirements: PRIV-01, PRIV-02, PRIV-03, PRIV-06, BKUP-01, BKUP-02, BKUP-03,

from pathlib import Path
from typing import AsyncGenerator

import pytest

from httpx import AsyncClient
from sqlalchemy import select, text

from app.domain.archive.models import RawArchiveItem
from app.domain.derived_memory.models import (

from app.domain.canonical_memory.models import CanonicalMemoryItem
from app.domain.audit.models import AuditEvent


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

async def _create_archive_item(
    session: AsyncSession,
    source_type: str = "text",
    raw_content: str = "integration test content",
) -> RawArchiveItem:
    item = RawArchiveItem(
        id=uuid.uuid4(),
        source_type=source_type,
        raw_content=raw_content,
        source_system="test",
    )
    session.add(item)
    await session.flush()
    return item

@pytest.mark.asyncio
async def test_priv01_delete_returns_204(client: AsyncClient, db_session_phase4: AsyncSession):
    """DELETE /api/archive/{id} returns 204 No Content."""
    item = await _create_archive_item(db_session_phase4)
    await db_session_phase4.commit()

    resp = await client.delete(f"/api/archive/{item.id}")
    assert resp.status_code == 204


@pytest.mark.asyncio
async def test_priv01_delete_sets_deleted_at(client: AsyncClient, db_session_phase4: AsyncSession):
    """After deletion, deleted_at is set on the archive item."""
    item = await _create_archive_item(db_session_phase4)
    await db_session_phase4.commit()

    await client.delete(f"/api/archive/{item.id}")

    await db_session_phase4.refresh(item)
    assert item.deleted_at is not None

# PRIV-06 — Auth token enforcement
# ---------------------------------------------------------------------------

@pytest.mark.asyncio
async def test_priv06_unauthenticated_delete_rejected_when_auth_enabled(
    client: AsyncClient, db_session_phase4: AsyncSession, monkeypatch: pytest.MonkeyPatch
):
    """When AUTH_TOKEN is set, DELETE without token returns 401."""
    monkeypatch.setenv("AUTH_TOKEN", "test-secret-token")

    item = await _create_archive_item(db_session_phase4)
    await db_session_phase4.commit()

    # Request without Authorization header
    resp = await client.delete(f"/api/archive/{item.id}")
    assert resp.status_code == 401


@pytest.mark.asyncio
async def test_priv06_authenticated_delete_succeeds(
    client: AsyncClient, db_session_phase4: AsyncSession, monkeypatch: pytest.MonkeyPatch
):
    """When AUTH_TOKEN is set, DELETE with correct Bearer token returns 204."""
    token = "test-secret-token"
    monkeypatch.setenv("AUTH_TOKEN", token)

    item = await _create_archive_item(db_session_phase4)
    await db_session_phase4.commit()

    resp = await client.delete(
        f"/api/archive/{item.id}",
        headers={"Authorization": f"Bearer {token}"},
    )
    assert resp.status_code == 204

)
async def test_bkup02_restore_accepts_valid_backup(client: AsyncClient):
    """BKUP-02: POST /api/backup/restore returns 200 with a valid backup filename."""
    # First create a backup
    create_resp = await client.post("/api/backup/create")
    if create_resp.status_code not in (200, 201):
        pytest.skip("Backup creation failed — skipping restore test")

    body = create_resp.json()
    filename = body.get("filename") or Path(body.get("path", "")).name
    if not filename:
        pytest.skip("No filename in backup response")

    restore_resp = await client.post("/api/backup/restore", json={"filename": filename})
    assert restore_resp.status_code == 200
