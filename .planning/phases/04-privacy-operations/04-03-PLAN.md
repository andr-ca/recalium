  """Telemetry service — increment daily usage counters.

  PORT-02: All telemetry stays local. Upsert pattern prevents N+1 selects.

  Event type → column mapping:
    "search"           → searches
    "retrieval"        → retrievals
    "mcp_retrieval"    → mcp_retrievals
    "ui_retrieval"     → ui_retrievals
    "fact_reviewed"    → facts_reviewed
    "canonical_created"→ canonical_created
  """
  from __future__ import annotations

  import logging
  from datetime import date, datetime, timezone, timedelta

  from sqlalchemy import text
  from sqlalchemy.ext.asyncio import AsyncSession

  logger = logging.getLogger(__name__)

  # Maps event_type string → telemetry column name
  _EVENT_TO_COLUMN: dict[str, str] = {
      "search": "searches",
      "retrieval": "retrievals",
      "mcp_retrieval": "mcp_retrievals",
      "ui_retrieval": "ui_retrievals",
      "fact_reviewed": "facts_reviewed",
      "canonical_created": "canonical_created",
  }


  async def increment_telemetry(
      session: AsyncSession,
      event_type: str,
      today: date | None = None,
  ) -> None:
      """Increment a daily usage counter using PostgreSQL upsert.

      Uses INSERT ... ON CONFLICT DO UPDATE — no SELECT first, no ORM object.
      Best-effort: logs a warning and returns on unknown event_type.
      """
      col = _EVENT_TO_COLUMN.get(event_type)
      if col is None:
          logger.warning("Unknown telemetry event_type: %s — skipping", event_type)
          return

      day = today or date.today()

      await session.execute(
          text(
              f"INSERT INTO telemetry (date, {col}) VALUES (:date, 1) "
              f"ON CONFLICT (date) DO UPDATE SET {col} = telemetry.{col} + 1"
          ),
          {"date": day},
      )
      # Note: caller is responsible for commit (FastAPI dependency handles this)


  async def get_telemetry_summary(
      session: AsyncSession,
      days: int = 30,
  ) -> list[dict]:
      """Return daily telemetry for the last N days, newest first.

      Returns list of dicts with keys: date, searches, retrievals,
      facts_reviewed, canonical_created, mcp_retrievals, ui_retrievals.
      Missing days (no data) are not included.
      """
      cutoff = date.today() - timedelta(days=days - 1)
      result = await session.execute(
          text(
              "SELECT date, searches, retrievals, facts_reviewed, "
              "canonical_created, mcp_retrievals, ui_retrievals "
              "FROM telemetry WHERE date >= :cutoff ORDER BY date DESC"
          ),
          {"cutoff": cutoff},
      )
      rows = result.fetchall()
      return [
          {
              "date": str(row.date),
              "searches": row.searches,
              "retrievals": row.retrievals,
              "facts_reviewed": row.facts_reviewed,
              "canonical_created": row.canonical_created,
              "mcp_retrievals": row.mcp_retrievals,
              "ui_retrievals": row.ui_retrievals,
          }
          for row in rows
      ]
  ```

  """Backup and restore service.

  BKUP-01: Scheduled daily backups with 30-day retention.
  BKUP-02: Restore within 15 minutes.
  BKUP-03: Restore recovers all data including audit events and configuration.
  PRIV-03: backup_predates_deletion() detects if a backup predates any deletion event.

  Tool: pg_dump (custom format -Fc) and pg_restore.
  Backup location: /backups/ (bind-mounted from ./backups on host).
  """
  from __future__ import annotations

  import asyncio
  import logging
  import os
  from datetime import datetime, timezone, timedelta
  from pathlib import Path

  from sqlalchemy import select, text
  from sqlalchemy.ext.asyncio import AsyncSession

  from app.infrastructure.settings import get_settings

  logger = logging.getLogger(__name__)

  DEFAULT_BACKUP_DIR = "/backups"
  BACKUP_RETENTION_DAYS = 30


  def _backup_filename(timestamp: datetime | None = None) -> str:
      """Generate backup filename from timestamp."""
      ts = timestamp or datetime.now(timezone.utc)
      return f"recalium_{ts.strftime('%Y%m%d_%H%M%S')}.dump"


  def _parse_backup_timestamp(filename: str) -> datetime | None:
      """Extract datetime from backup filename. Returns None if unparseable."""
      try:
          # Format: recalium_YYYYMMDD_HHMMSS.dump
          stem = Path(filename).stem  # recalium_20260101_120000
          parts = stem.split("_")
          if len(parts) != 3:
              return None
          date_str = parts[1]
          time_str = parts[2]
          dt_str = f"{date_str}_{time_str}"
          return datetime.strptime(dt_str, "%Y%m%d_%H%M%S").replace(tzinfo=timezone.utc)
      except (ValueError, IndexError):
          return None


  async def create_backup(backup_dir: str = DEFAULT_BACKUP_DIR) -> dict:
      """Run pg_dump and save backup to backup_dir.

      Returns dict with: filename, created_at (ISO), size_bytes, backup_dir.
      Raises RuntimeError on pg_dump failure.
      """
      settings = get_settings()
      Path(backup_dir).mkdir(parents=True, exist_ok=True)

      now = datetime.now(timezone.utc)
      filename = _backup_filename(now)
      filepath = Path(backup_dir) / filename

      env = os.environ.copy()
      env["PGPASSWORD"] = settings.postgres_password

      cmd = [
          "pg_dump",
          "-Fc",          # Custom format (compressed, restorable with pg_restore)
          "-h", settings.postgres_host,
          "-p", str(settings.postgres_port),
          "-U", settings.postgres_user,
          "-d", settings.postgres_db,
          "-f", str(filepath),
      ]

      logger.info("Starting backup: %s", filename)
      proc = await asyncio.create_subprocess_exec(
          *cmd,
          env=env,
          stdout=asyncio.subprocess.PIPE,
          stderr=asyncio.subprocess.PIPE,
      )
      stdout, stderr = await proc.communicate()

      if proc.returncode != 0:
          error_msg = stderr.decode().strip()
          logger.error("pg_dump failed (rc=%d): %s", proc.returncode, error_msg)
          raise RuntimeError(f"pg_dump failed: {error_msg}")

      size_bytes = filepath.stat().st_size
      logger.info("Backup complete: %s (%d bytes)", filename, size_bytes)

      return {
          "filename": filename,
          "created_at": now.isoformat(),
          "size_bytes": size_bytes,
          "backup_dir": backup_dir,
      }


  async def list_backups(backup_dir: str = DEFAULT_BACKUP_DIR) -> list[dict]:
      """List all backup files in backup_dir, newest first.

      Returns list of dicts: filename, created_at, size_bytes.
      Does NOT include has_post_deletion_events (requires DB — caller enriches).
      """
      backup_path = Path(backup_dir)
      if not backup_path.exists():
          return []

      backups = []
      for f in backup_path.glob("recalium_*.dump"):
          ts = _parse_backup_timestamp(f.name)
          backups.append({
              "filename": f.name,
              "created_at": ts.isoformat() if ts else None,
              "size_bytes": f.stat().st_size,
          })

      # Sort newest first
      backups.sort(key=lambda b: b["created_at"] or "", reverse=True)
      return backups


  async def restore_backup(
      filename: str,
      backup_dir: str = DEFAULT_BACKUP_DIR,
  ) -> dict:
      """Restore a backup using pg_restore.

      BKUP-02: Intended to complete within 15 minutes for personal-scale databases.
      Uses --clean --if-exists --no-owner --no-privileges for safe restore.
      """
      settings = get_settings()
      filepath = Path(backup_dir) / filename

      if not filepath.exists():
          raise FileNotFoundError(f"Backup file not found: {filename}")

      env = os.environ.copy()
      env["PGPASSWORD"] = settings.postgres_password

      cmd = [
          "pg_restore",
          "--clean",
          "--if-exists",
          "--no-owner",
          "--no-privileges",
          "-h", settings.postgres_host,
          "-p", str(settings.postgres_port),
          "-U", settings.postgres_user,
          "-d", settings.postgres_db,
          str(filepath),
      ]

      logger.info("Starting restore from: %s", filename)
      proc = await asyncio.create_subprocess_exec(
          *cmd,
          env=env,
          stdout=asyncio.subprocess.PIPE,
          stderr=asyncio.subprocess.PIPE,
      )

      try:
          stdout, stderr = await asyncio.wait_for(proc.communicate(), timeout=900)  # 15min SLA
      except asyncio.TimeoutError:
          proc.kill()
          raise RuntimeError("pg_restore timed out after 15 minutes")

      if proc.returncode not in (0, 1):  # pg_restore rc=1 is warnings (acceptable)
          error_msg = stderr.decode().strip()
          logger.error("pg_restore failed (rc=%d): %s", proc.returncode, error_msg)
          raise RuntimeError(f"pg_restore failed: {error_msg}")

      logger.info("Restore complete from: %s", filename)
      return {"status": "restored", "filename": filename}


  async def delete_old_backups(
      backup_dir: str = DEFAULT_BACKUP_DIR,
      retention_days: int = BACKUP_RETENTION_DAYS,
  ) -> int:
      """Delete backup files older than retention_days.

      Returns number of files deleted.
      """
      backup_path = Path(backup_dir)
      if not backup_path.exists():
          return 0

      cutoff = datetime.now(timezone.utc) - timedelta(days=retention_days)
      deleted = 0

      for f in backup_path.glob("recalium_*.dump"):
          ts = _parse_backup_timestamp(f.name)
          if ts and ts < cutoff:
              f.unlink()
              logger.info("Deleted old backup: %s", f.name)
              deleted += 1

      return deleted


  async def backup_predates_deletion(
      session: AsyncSession,
      backup_timestamp: datetime,
  ) -> bool:
      """Check if any archive_delete audit event occurred after backup_timestamp.

      PRIV-03: UI uses this to warn when a backup may contain since-deleted data.
      Returns True if there are deletion events more recent than the backup.
      """
      result = await session.execute(
          text(
              "SELECT 1 FROM audit_events "
              "WHERE event_type = 'archive_delete' AND occurred_at > :ts "
              "LIMIT 1"
          ),
          {"ts": backup_timestamp},
      )
      return result.scalar_one_or_none() is not None
  ```
