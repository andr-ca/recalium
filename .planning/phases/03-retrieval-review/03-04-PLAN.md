"""Canonical memory service.

CANM-01: Users can create, edit, delete, mark-disputed/stale canonical memory items.
CANM-02: Canonical items ranked first in retrieval (enforced in retrieval service).
CANM-03: Promotion only via explicit user action — never auto-promoted.
CANM-04: Facts without source_span require confirmed=True.
"""
from __future__ import annotations

import logging
import uuid
from datetime import datetime, timezone

from sqlalchemy import select, update
from sqlalchemy.ext.asyncio import AsyncSession

from app.domain.canonical_memory.models import CanonicalMemoryItem
from app.domain.derived_memory.models import Fact

logger = logging.getLogger(__name__)


class CanonicalMemoryError(Exception):
    """Raised for canonical memory rule violations."""


async def promote_fact(
    session: AsyncSession,
    fact_id: uuid.UUID,
    promoted_by: str = "user_ui",
    provenance_note: str | None = None,
    confirmed: bool = False,
) -> CanonicalMemoryItem:
    """Promote a Fact to canonical memory.

    CANM-03: Explicit user action only — never called by automated pipeline.
    CANM-04: If fact.source_span is empty, confirmed=True must be passed.

    Raises CanonicalMemoryError if:
    - fact not found
    - fact.source_span is empty and confirmed is False
    """
    result = await session.execute(
        select(Fact).where(Fact.id == fact_id, Fact.source_status == "active")
    )
    fact = result.scalars().first()
    if fact is None:
        raise CanonicalMemoryError(f"Fact {fact_id} not found or not active")

    # CANM-04: Empty source_span requires explicit confirmation
    if not fact.source_span.strip() and not confirmed:
        raise CanonicalMemoryError(
            "Fact has empty source_span — pass confirmed=True to promote it to canonical memory (CANM-04)"
        )

    item = CanonicalMemoryItem(
        id=uuid.uuid4(),
        raw_archive_id=fact.raw_archive_id,
        fact_id=fact.id,
        content=fact.fact_text,
        status="active",
        source_status="active",
        promoted_from="fact",
        promoted_by=promoted_by,
        provenance_note=provenance_note,
    )
    session.add(item)
    await session.commit()
    await session.refresh(item)
    logger.info("Promoted fact %s to canonical memory item %s", fact_id, item.id)
    return item


async def create_canonical(
    session: AsyncSession,
    content: str,
    raw_archive_id: uuid.UUID | None = None,
    promoted_by: str = "user_ui",
    provenance_note: str | None = None,
) -> CanonicalMemoryItem:
    """Create a canonical memory item manually (not from a fact).

    CANM-03: Explicit user action only.
    """
    if not content.strip():
        raise CanonicalMemoryError("content must not be empty")

    item = CanonicalMemoryItem(
        id=uuid.uuid4(),
        raw_archive_id=raw_archive_id,
        fact_id=None,
        content=content,
        status="active",
        source_status="active",
        promoted_from="manual",
        promoted_by=promoted_by,
        provenance_note=provenance_note,
    )
    session.add(item)
    await session.commit()
    await session.refresh(item)
    return item


async def list_canonical(
    session: AsyncSession,
    include_non_active: bool = False,
    limit: int = 100,
    offset: int = 0,
) -> list[CanonicalMemoryItem]:
    """List canonical memory items.

    By default returns only source_status='active' AND status='active' items.
    Set include_non_active=True to include disputed/stale items.
    """
    stmt = select(CanonicalMemoryItem).where(
        CanonicalMemoryItem.source_status == "active"
    )
    if not include_non_active:
        stmt = stmt.where(CanonicalMemoryItem.status == "active")
    stmt = stmt.order_by(CanonicalMemoryItem.created_at.desc()).limit(limit).offset(offset)
    result = await session.execute(stmt)
    return list(result.scalars().all())


async def get_canonical(
    session: AsyncSession,
    item_id: uuid.UUID,
) -> CanonicalMemoryItem | None:
    """Get a single canonical memory item by ID."""
    result = await session.execute(
        select(CanonicalMemoryItem).where(CanonicalMemoryItem.id == item_id)
    )
    return result.scalars().first()


async def update_canonical(
    session: AsyncSession,
    item_id: uuid.UUID,
    content: str | None = None,
    status: str | None = None,
    provenance_note: str | None = None,
) -> CanonicalMemoryItem:
    """Update a canonical memory item.

    CANM-01: Users can edit content, mark disputed or stale.
    Allowed status values: "active" | "disputed" | "stale" | "source_removed_review_required"
    """
    item = await get_canonical(session, item_id)
    if item is None:
        raise CanonicalMemoryError(f"CanonicalMemoryItem {item_id} not found")

    allowed_statuses = {"active", "disputed", "stale", "source_removed_review_required"}
    if status is not None and status not in allowed_statuses:
        raise CanonicalMemoryError(f"Invalid status: {status!r}. Must be one of {allowed_statuses}")

    if content is not None:
        if not content.strip():
            raise CanonicalMemoryError("content must not be empty")
        item.content = content
    if status is not None:
        item.status = status
    if provenance_note is not None:
        item.provenance_note = provenance_note

    item.updated_at = datetime.now(timezone.utc)
    await session.commit()
    await session.refresh(item)
    return item


async def delete_canonical(
    session: AsyncSession,
    item_id: uuid.UUID,
) -> None:
    """Hard-delete a canonical memory item (CANM-01).

    For soft-suppression use update_canonical(status='stale').
    """
    item = await get_canonical(session, item_id)
    if item is None:
        raise CanonicalMemoryError(f"CanonicalMemoryItem {item_id} not found")
    await session.delete(item)
    await session.commit()
    logger.info("Deleted canonical memory item %s", item_id)

"""Review queue service.

CANM-05: Groups duplicate/overlapping/contradicting facts for manageable cleanup.
Items are materialized from conflict_groups.
Status transitions: pending → resolved | dismissed.
"""
from __future__ import annotations

import logging
import uuid
from datetime import datetime, timezone

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.domain.derived_memory.models import ConflictGroup
from app.domain.review_queue.models import ReviewQueueItem

logger = logging.getLogger(__name__)


class ReviewQueueError(Exception):
    """Raised for review queue rule violations."""


async def backfill_review_queue(session: AsyncSession) -> int:
    """Create ReviewQueueItems for any ConflictGroups without a queue entry.

    Called on Phase 3 first startup and optionally after each conflict detection run.
    Returns the number of new items created.
    """
    # Find conflict groups that have no review_queue_item yet
    result = await session.execute(
        select(ConflictGroup).where(
            ConflictGroup.source_status == "active",
            ~ConflictGroup.id.in_(
                select(ReviewQueueItem.conflict_group_id)
            ),
        )
    )
    groups = result.scalars().all()

    created = 0
    for group in groups:
        item = ReviewQueueItem(
            id=uuid.uuid4(),
            conflict_group_id=group.id,
            item_type=group.group_type,
            status="pending",
            source_status="active",
        )
        session.add(item)
        created += 1

    if created > 0:
        await session.commit()
        logger.info("Backfilled %d review queue items from conflict groups", created)

    return created


async def list_review_queue(
    session: AsyncSession,
    status_filter: str | None = "pending",
    limit: int = 100,
    offset: int = 0,
) -> list[ReviewQueueItem]:
    """List review queue items.

    Defaults to status='pending' items. Pass status_filter=None for all items.
    Always filters source_status='active'.
    """
    stmt = select(ReviewQueueItem).where(ReviewQueueItem.source_status == "active")
    if status_filter is not None:
        stmt = stmt.where(ReviewQueueItem.status == status_filter)
    stmt = stmt.order_by(ReviewQueueItem.created_at.asc()).limit(limit).offset(offset)
    result = await session.execute(stmt)
    return list(result.scalars().all())


async def _get_item(session: AsyncSession, item_id: uuid.UUID) -> ReviewQueueItem:
    result = await session.execute(
        select(ReviewQueueItem).where(ReviewQueueItem.id == item_id)
    )
    item = result.scalars().first()
    if item is None:
        raise ReviewQueueError(f"ReviewQueueItem {item_id} not found")
    return item


async def resolve_item(
    session: AsyncSession,
    item_id: uuid.UUID,
    resolved_by: str = "user_ui",
    resolution_note: str | None = None,
) -> ReviewQueueItem:
    """Mark a review queue item as resolved.

    Transition: pending → resolved. Idempotent if already resolved.
    """
    item = await _get_item(session, item_id)
    if item.status == "dismissed":
        raise ReviewQueueError(f"Cannot resolve item {item_id} — it is already dismissed")
    item.status = "resolved"
    item.resolved_by = resolved_by
    item.resolution_note = resolution_note
    item.resolved_at = datetime.now(timezone.utc)
    await session.commit()
    await session.refresh(item)
    return item


async def dismiss_item(
    session: AsyncSession,
    item_id: uuid.UUID,
    resolved_by: str = "user_ui",
    resolution_note: str | None = None,
) -> ReviewQueueItem:
    """Dismiss a review queue item (not a conflict after all).

    Transition: pending → dismissed. Idempotent if already dismissed.
    """
    item = await _get_item(session, item_id)
    if item.status == "resolved":
        raise ReviewQueueError(f"Cannot dismiss item {item_id} — it is already resolved")
    item.status = "dismissed"
    item.resolved_by = resolved_by
    item.resolution_note = resolution_note
    item.resolved_at = datetime.now(timezone.utc)
    await session.commit()
    await session.refresh(item)
    return item
