# ── SRCH-04: Budget trimming follows canonical→fact→summary→excerpt order ─────

def test_srch04_budget_trimming_priority_order():
    """SRCH-04: Budget trimming uses canonical→fact→summary→excerpt priority."""
    items = [
        RetrievalItem(id="1", type="excerpt", content="A" * 100, score=0.9, source_id="s1", source_system="x", captured_at="", conflict_label=None, provenance={}),
        RetrievalItem(id="2", type="canonical", content="B" * 100, score=0.5, source_id="s2", source_system="x", captured_at="", conflict_label=None, provenance={}),
        RetrievalItem(id="3", type="fact", content="C" * 100, score=0.7, source_id="s3", source_system="x", captured_at="", conflict_label=None, provenance={}),
        RetrievalItem(id="4", type="summary", content="D" * 100, score=0.8, source_id="s4", source_system="x", captured_at="", conflict_label=None, provenance={}),
    ]
    # Budget fits canonical + fact but not summary + excerpt
    trimmed, used, reason = apply_budget_trimming(items, budget=250)
    types = [i.type for i in trimmed]
    # canonical must come before fact, fact before summary
    assert types.index("canonical") < types.index("fact")
    assert used <= 250


def test_srch04_budget_trimming_no_mid_item_truncation():
    """SRCH-04: Budget trimming never truncates mid-item — skips if doesn't fit."""
    items = [
        RetrievalItem(id="1", type="canonical", content="A" * 50, score=0.9, source_id="s1", source_system="x", captured_at="", conflict_label=None, provenance={}),
        RetrievalItem(id="2", type="fact", content="B" * 100, score=0.8, source_id="s2", source_system="x", captured_at="", conflict_label=None, provenance={}),
    ]
    # Budget fits canonical (50) but not fact (100). Fact must be skipped, not truncated.
    trimmed, used, reason = apply_budget_trimming(items, budget=80)
    assert len(trimmed) == 1
    assert trimmed[0].type == "canonical"
    assert len(trimmed[0].content) == 50  # not truncated
